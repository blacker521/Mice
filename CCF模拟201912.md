#include <iostream>
using namespace std;
int cnt = 0,num = 0,n,a[4];

bool check(int num){
    if (num % 7 == 0) return true;
    while (num > 0) {
        if (num % 10 == 7) return true;
        num /= 10;
    }
    return false;;
}

int main() {
    scanf("%d",&n);
    while (cnt < n) {
        num ++;
        if (check(num)) {a[(num - 1) % 4] ++;cnt --;}
        cnt ++;
    }
    printf("%d\n%d\n%d\n%d\n",a[0],a[1],a[2],a[3]);
    return 0;
}
